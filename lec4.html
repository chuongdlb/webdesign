3. ü§î Conditional Statements

3.1 If-Else Statement

let score = 85;

if (score >= 90) {
  console.log("A grade - Excellent!");
} else if (score >= 80) {
  console.log("B grade - Good job!");
} else if (score >= 70) {
  console.log("C grade - Keep trying!");
} else {
  console.log("Need improvement");
}

3.2 Switch Statement

let day = "Monday";

switch (day) {
  case "Monday":
    console.log("Start of work week");
    break;
  case "Friday":
    console.log("TGIF!");
    break;
  case "Saturday":
  case "Sunday":
    console.log("Weekend!");
    break;
  default:
    console.log("Regular day");
}

4. üîÑ Loop Statements

4.1 For Loop

// Basic for loop
for (let i = 1; i <= 5; i++) {
  console.log("Count: " + i);
}

// For loop with arrays
let colors = ["red", "green", "blue"];
for (let i = 0; i < colors.length; i++) {
  console.log(colors[i]);
}

4.2 While Loop

let count = 0;
while (count < 3) {
  console.log("While loop: " + count);
  count++;
}

4.3 For...of and For...in Loops

// For...of (values)
let animals = ["cat", "dog", "bird"];
for (let animal of animals) {
  console.log(animal);
}

// For...in (properties)
let person = {name: "Alice", age: 25};
for (let key in person) {
  console.log(key + ": " + person[key]);
}
